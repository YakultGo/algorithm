### 数组写法,当字符串太多的时候,会溢出

```c++
#define MAX 100000
struct trie {
    int nex[MAX+1][26], cnt;
    bool exist[MAX+1];  // 该结点结尾的字符串是否存在
    void insert(const string& s) {  // 插入字符串
        int p = 0;
        for (char i : s) {
            int c = i - 'a';
            if (!nex[p][c])
                nex[p][c] = ++cnt;  // 如果没有，就添加结点
            p = nex[p][c];
        }
        exist[p] = true;
    }
    bool search(const string& s) {  // 查找字符串
        int p = 0;
        for (char i : s) {
            int c = i - 'a';
            if (!nex[p][c])
                return false;
            p = nex[p][c];
        }
        return exist[p];
    }
};
```

### 指针写法

```c++
struct trie {
    struct trie *nex[26]{};
    int tag;
    trie(){
        memset(nex, 0, sizeof(nex));
        tag = 0;
    }
    void insert(const string &s){
        trie *p = this;
        for (char i : s) {
            if (p->nex[i - 'a'] == nullptr) {
                p->nex[i - 'a'] = new trie();
            }
            p = p->nex[i - 'a'];
        }
        p->tag = 1;
    }
    bool search(const string &s){
        trie *p = this;
        for (char i : s) {
            if (p->nex[i - 'a'] == nullptr) {
                return false;
            }
            p = p->nex[i - 'a'];
        }
        return p->tag;
    }
    bool startsWith(string prefix) {
          trie *p = this;
          for(char i : prefix) {
              if(p->nex[i-'a'] == nullptr)
                  return false;
              p = p->nex[i-'a'];
          }
          return true;
    }
};
```

### 如果加了通配符,则利用dfs

```c++
struct trie {
    struct trie *nex[26]{};
    int tag;
    trie(){
        memset(nex, 0, sizeof(nex));
        tag = 0;
    }
    void insert(const string &s){
        trie *p = this;
        for (char i : s) {
            if (p->nex[i - 'a'] == nullptr) {
                p->nex[i - 'a'] = new trie();
            }
            p = p->nex[i - 'a'];
        }
        p->tag = 1;
    }
    bool search(const string &s){
        trie *p = this;
        dfs(s,0,p);
    }
    bool dfs(const string &word, int index, trie* node) {
        if(index == word.size())
            return node->tag;
        char c = word[index];
        if(c == '.') {
            for(int i = 0; i < 26; i++) {
                trie* child = node->nex[i];
                if(child != nullptr && dfs(word,index+1,child))
                    return true;
            }
        }else {
            trie* child = node->nex[c-'a'];
            if(child != nullptr && dfs(word,index+1,child))
                return true;
        }
        return false;
    }
};
```

### 0-1trie，计算最大异或

```c++
struct trie {
    struct trie *nxt[2]{};
    int tag;
    trie() : tag(0) {
        memset(nxt, 0, sizeof(nxt));
    }
    void insert(int x) {
        trie *p = this;
        for (int i = 30; i >= 0; i--) {
            int bit = (x >> i) & 1;
            if (p->nxt[bit] == nullptr)
                p->nxt[bit] = new trie();
            p = p->nxt[bit];
            p->tag++;
        }
    }
    void erase(int x) {
        trie *p = this;
        for (int i = 30; i >= 0; i--) {
            int bit = (x >> i) & 1;
            p = p->nxt[bit];
            p->tag--;
        }
    }
    // 获取最大异或结果
    int getVal(int x) {
        int ans = 0;
        trie *p = this;
        for (int i = 30; i >= 0; i--) {
            int a = (x >> i) & 1, b = 1 - a;
            if (p->nxt[b] != nullptr && p->nxt[b]->tag > 0) {
                ans |= (b << i);
                p = p->nxt[b];
            } else {
                ans |= (a << i);
                p = p->nxt[a];
            }
        }
        return x ^ ans;
    }
};
```
