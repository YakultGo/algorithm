vector<bool> isPrime(MAXX+10,true);
vector<int> primes;

    isPrime[1] = false;

### 线性筛

```c++
#define MAXX ((int) 1e5)
vector<bool> isPrime(MAXX+10,true);
vector<int> primes;
int init = []() ->int {
    isPrime[1] = false;
    for (int i = 2; i <= MAXX; ++i) {
        if (isPrime[i]) {
            primes.push_back(i);
        }
        for (int j = 0; j < primes.size() && i * primes[j] <= MAXX; ++j) {
            isPrime[i * primes[j]] = false;
            if (i % primes[j] == 0) {
                break;
            }
        }
    }
    return 0;
}();
```

### 将埃式筛改造一下，就可以用来求质因数的个数和对应的质因数

```c++
#define MAXX ((int)2e5)
vector<int> f(MAXX + 10, 0);
vector<int> arr[MAXX + 10];
int init = []() -> int {
    f[1] = 1;
    for (int i = 2; i <= MAXX; i++)
        if (!f[i]) {
            for (int j = i; j <= MAXX; j += i) {
                f[j]++;
                // arr[j].push_back(i);
                int z = j;
                while (z && z % i == 0) {
                    arr[j].push_back(i);
                    z /= i;
                }
            }
        }
    return 0;
}();
```
