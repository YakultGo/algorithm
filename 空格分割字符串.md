```c++
		string input = "This is a test string";
    istringstream iss(input); // 将输入字符串包装成一个字符串流

    vector<string> tokens; // 用于存储分割后的子串

    string token;
    while (iss >> token) {
        tokens.push_back(token); // 将分割后的子串添加到向量中
    }

    // 打印分割后的子串
    for (const string &subStr : tokens) {
        cout << subStr << endl;
    }
```
