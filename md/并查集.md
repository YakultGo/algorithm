### 普通并查集

```c++
struct dsu {
    vector<int> pa, size;
    explicit dsu(int size_) : pa(size_), size(size_, 1) {
        iota(pa.begin(), pa.end(), 0);
    }
    void unite(int x, int y) {
        x = find(x), y = find(y);
        if (x == y) return;
        if (size[x] < size[y]) swap(x, y);
        pa[y] = x;
        size[x] += size[y];
    }
    int find(int x) {
        return pa[x] == x ? x : pa[x] = find(pa[x]);
    }
};
```

vector<int>fa(n+1);
for (int i = 1; i <= n; ++i) fa[i] = i;
function<int(int)> find = [&] (int x) ->int {
    int t = x;
    while (fa[x] != x)	// 查找父节点
        x = fa[x];
    while (t != x) {	// 路径压缩
        int tmp = fa[t];
        fa[t] = x;
        t = tmp;
    }
    return x;
};

### 连通分量并查集

```c++
struct dsu {
        vector<int> pa, size;
        int count;
        explicit dsu(int size_) : pa(size_), size(size_, 1),count(0) {
            iota(pa.begin(), pa.end(), 0);
        }
        void unite(int x, int y) {
            x = find(x), y = find(y);
            if (x == y) return;
            if (size[x] < size[y]) 
                swap(x, y);
            pa[y] = x;
            size[x] += size[y];
            count--;
        }
        int find(int x) {
            return pa[x] == x ? x : pa[x] = find(pa[x]);
        }
    };
```
